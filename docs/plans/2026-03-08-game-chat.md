**Step 1: Create `src/chat/mod.rs`**

```rust
use bevy::prelude::*;

/// Category of a chat message, determines prefix and color.
#[derive(Debug, Clone)]
pub enum MessageCategory {
    System,
    Dialog { speaker: String },
    PlayerCommand,
    PlayerChat,
}

/// A single chat message.
#[derive(Debug, Clone)]
pub struct ChatMessage {
    pub text: String,
    pub category: MessageCategory,
    pub timestamp: f64,
}

/// Event fired when the player enters a `/command`.
#[derive(Event, Debug, Clone)]
pub struct ChatCommandEvent {
    pub command: String,
    pub args: Vec<String>,
}

/// Global chat state resource.
#[derive(Resource)]
pub struct ChatState {
    pub messages: Vec<ChatMessage>,
    pub is_active: bool,
    pub input_buffer: String,
    pub scroll_offset: i32,
    pub max_messages: usize,
}

impl ChatState {
    pub fn new(max_messages: usize) -> Self {
        Self {
            messages: Vec::new(),
            is_active: false,
            input_buffer: String::new(),
            scroll_offset: 0,
            max_messages,
        }
    }

    /// Add a system message.
    pub fn send_system(&mut self, text: &str, time: f64) {
        self.push(text.to_string(), MessageCategory::System, time);
    }

    /// Add an NPC dialog message.
    pub fn send_dialog(&mut self, speaker: &str, text: &str, time: f64) {
        self.push(
            text.to_string(),
            MessageCategory::Dialog { speaker: speaker.to_string() },
            time,
        );
    }

    fn push(&mut self, text: String, category: MessageCategory, timestamp: f64) {
        self.messages.push(ChatMessage {
            text,
            category,
            timestamp,
        });
        if self.messages.len() > self.max_messages {
            self.messages.remove(0);
        }
        // Reset scroll to bottom on new message
        self.scroll_offset = 0;
    }
}

pub struct ChatPlugin;

impl Plugin for ChatPlugin {
    fn build(&self, app: &mut App) {
        app.add_event::<ChatCommandEvent>();
    }
}
```

/// Marker for the chat root container.
#[derive(Component)]
pub struct ChatRoot;

/// Marker for the messages scroll area.
#[derive(Component)]
pub struct ChatMessagesArea;

/// Marker for the input text field.

#[derive(Component)]

pub fn spawn_chat(commands: &mut Commands, theme: &UiTheme) {
    let chat = &theme.chat;

    commands
        .spawn((
            ChatRoot,
            Node {
                position_type: PositionType::Absolute,
                right: Val::Px(8.0),
                bottom: Val::Px(8.0),
                width: Val::Px(chat.width),
                height: Val::Px(chat.height),
                flex_direction: FlexDirection::Column,
                justify_content: JustifyContent::FlexEnd,
                ..default()
            },
            Pickable::IGNORE,
        ))
        .with_children(|parent| {
            // Background
            parent.spawn((
                ChatBackground,
                Node {
                    position_type: PositionType::Absolute,
                    width: Val::Percent(100.0),
                    height: Val::Percent(100.0),
                    ..default()
                },
                BackgroundColor(Color::NONE),
                Pickable::IGNORE,
            ));

            // Messages area (scrollable, fills available space)
            parent.spawn((
                ChatMessagesArea,
                Node {
                    flex_direction: FlexDirection::Column,
                    justify_content: JustifyContent::FlexEnd,
                    flex_grow: 1.0,
                    overflow: Overflow::clip_y(),
                    padding: UiRect::all(Val::Px(4.0)),
                    ..default()
                },
                Pickable::IGNORE,
            ));

            // Input line (hidden by default)
            parent.spawn((
                ChatInputLine,
                Text::new(""),
                TextFont {
                    font_size: chat.font_size,
                    ..default()
                },
                TextColor(Color::WHITE),
                Node {
                    padding: UiRect::axes(Val::Px(4.0), Val::Px(2.0)),
                    ..default()
                },
                BackgroundColor(chat.input_bg_color.clone().into()),
                Visibility::Hidden,
                Pickable::IGNORE,
            ));
        });
}

### Task 4: Chat input handling

**Files:**
- Modify: `src/ui/game_ui/chat.rs`
- Modify: `src/ui/game_ui/mod.rs`

**Step 1: Add chat input system**

System `chat_input_system` runs in `GameSet::Ui`:
- If NOT active: Enter key → set `is_active = true`, show input line, show background
- If active:
  - Read `ReceivedCharacter` events for text input (or use `KeyCode` mapping for basic ASCII)
  - Backspace → remove last char
  - Enter → process input (parse command or add as PlayerChat message), deactivate
  - Escape → clear buffer, deactivate
  - Update input line text entity

```rust
pub fn chat_input_system(
    keys: Res<ButtonInput<KeyCode>>,
    mut char_events: EventReader<KeyboardInput>,
    mut chat_state: ResMut<ChatState>,
    mut chat_command_events: EventWriter<ChatCommandEvent>,
    time: Res<Time>,
    mut input_line: Query<(&mut Text, &mut Visibility, &mut BackgroundColor), With<ChatInputLine>>,
    mut bg_query: Query<&mut BackgroundColor, (With<ChatBackground>, Without<ChatInputLine>)>,
    theme: Res<UiTheme>,
) {
    // Toggle active on Enter when not active
    if !chat_state.is_active {
        if keys.just_pressed(KeyCode::Enter) {
            chat_state.is_active = true;
            // Show input line and background
            for (mut text, mut vis, mut bg) in &mut input_line {
                *vis = Visibility::Visible;
                *bg = BackgroundColor(theme.chat.input_bg_color.clone().into());
                *text = Text::new("> _");
            }
            for mut bg in &mut bg_query {
                *bg = BackgroundColor(theme.chat.active_bg_color.clone().into());
            }
        }
        return;
    }

    // Active mode: handle keyboard input
    for event in char_events.read() {
        if event.state != bevy::input::ButtonState::Pressed {
            continue;
        }
        match event.key_code {
            KeyCode::Enter => {
                let input = chat_state.input_buffer.trim().to_string();
                if !input.is_empty() {
                    if input.starts_with('/') {
                        // Parse command
                        let parts: Vec<&str> = input[1..].split_whitespace().collect();
                        if let Some(cmd) = parts.first() {
                            let args: Vec<String> = parts[1..].iter().map(|s| s.to_string()).collect();
                            chat_command_events.write(ChatCommandEvent {
                                command: cmd.to_string(),
                                args,
                            });
                            chat_state.push(
                                input.clone(),
                                MessageCategory::PlayerCommand,
                                time.elapsed_secs_f64(),
                            );
                        }
                    } else {
                        chat_state.push(
                            input.clone(),
                            MessageCategory::PlayerChat,
                            time.elapsed_secs_f64(),
                        );
                    }
                }
                deactivate_chat(&mut chat_state, &mut input_line, &mut bg_query);
                return;
            }
            KeyCode::Escape => {
                deactivate_chat(&mut chat_state, &mut input_line, &mut bg_query);
                return;
            }
            KeyCode::Backspace => {
                chat_state.input_buffer.pop();
            }
            _ => {
                // Convert key event to character using logical_key
                if let bevy::input::keyboard::Key::Character(ref ch) = event.logical_key {
                    chat_state.input_buffer.push_str(ch.as_str());
                } else if event.key_code == KeyCode::Space {
                    chat_state.input_buffer.push(' ');
                }
            }
        }
    }

    // Update input line text
    for (mut text, _, _) in &mut input_line {
        *text = Text::new(format!("> {}_", chat_state.input_buffer));
    }
}

fn deactivate_chat(
    chat_state: &mut ResMut<ChatState>,
    input_line: &mut Query<(&mut Text, &mut Visibility, &mut BackgroundColor), With<ChatInputLine>>,
    bg_query: &mut Query<&mut BackgroundColor, (With<ChatBackground>, Without<ChatInputLine>)>,
) {
    chat_state.is_active = false;
    chat_state.input_buffer.clear();
    for (mut text, mut vis, _) in input_line.iter_mut() {
        *vis = Visibility::Hidden;
        *text = Text::new("");
    }
    for mut bg in bg_query.iter_mut() {
        *bg = BackgroundColor(Color::NONE);
    }
}
```

**Step 1: Add message rendering system**

System `chat_render_messages` checks if ChatState changed, rebuilds message text entities in `ChatMessagesArea`:

```rust
pub fn chat_render_messages(
    chat_state: Res<ChatState>,
    theme: Res<UiTheme>,
    time: Res<Time>,
    mut commands: Commands,
    area_query: Query<Entity, With<ChatMessagesArea>>,
    existing: Query<Entity, With<ChatMessageText>>,
) {
    let Ok(area) = area_query.single() else { return };

    // Despawn old messages
    for entity in &existing {
        commands.entity(entity).despawn();
    }

    let chat = &theme.chat;
    let now = time.elapsed_secs_f64();

    // Determine which messages to show
    let messages = if chat_state.is_active {
        // Show all messages (with scroll offset)
        let len = chat_state.messages.len();
        let start = (len as i32 - chat.visible_lines as i32 * 4 - chat_state.scroll_offset)
            .max(0) as usize;
        &chat_state.messages[start..]
    } else {
        // Show only recent messages (last visible_lines)
        let len = chat_state.messages.len();
        let start = len.saturating_sub(chat.visible_lines);
        &chat_state.messages[start..]
    };

    commands.entity(area).with_children(|parent| {
        for (i, msg) in messages.iter().enumerate() {
            let (prefix, color) = match &msg.category {
                MessageCategory::System => ("[System] ".to_string(), chat.system_color.clone()),
                MessageCategory::Dialog { speaker } => {
                    (format!("[{}] ", speaker), chat.dialog_color.clone())
                }
                MessageCategory::PlayerCommand => ("[Cmd] ".to_string(), chat.command_color.clone()),
                MessageCategory::PlayerChat => ("[You] ".to_string(), chat.player_color.clone()),
            };

            // Calculate opacity for fade in inactive mode
            let alpha = if chat_state.is_active {
                1.0
            } else {
                let age = (now - msg.timestamp) as f32;
                if age < chat.fade_delay_secs {
                    1.0
                } else {
                    let fade_progress = (age - chat.fade_delay_secs) / chat.fade_duration_secs;
                    (1.0 - fade_progress).clamp(0.0, 1.0)
                }
            };

            if alpha <= 0.0 && !chat_state.is_active {
                continue;
            }

            let mut text_color: Color = color.into();
            text_color.set_alpha(alpha);

            parent.spawn((
                ChatMessageText { index: i },
                Text::new(format!("{}{}", prefix, msg.text)),
                TextFont {
                    font_size: chat.font_size,
                    ..default()
                },
                TextColor(text_color),
                Pickable::IGNORE,
            ));
        }
    });
}
```

System `chat_scroll_system` reads mouse wheel events when chat is active and adjusts `scroll_offset`:

```rust
pub fn chat_scroll_system(
    mut chat_state: ResMut<ChatState>,
    mut scroll_events: EventReader<bevy::input::mouse::MouseWheel>,
) {
    if !chat_state.is_active {
        return;
    }

    for event in scroll_events.read() {
        let delta = event.y.signum() as i32;
        chat_state.scroll_offset = (chat_state.scroll_offset + delta)
            .max(0)
            .min(chat_state.messages.len().saturating_sub(1) as i32);
    }
}
```
